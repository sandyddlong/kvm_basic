```
      <mac address='52:54:00:d3:01:ac'/>
```

```
echo "52:54:$(dd if=/dev/urandom count=1 2>/dev/null | md5sum | sed -r 's/^(..)(..)(..)(..).*$/\1:\2:\3:\4/')"
```
